import React, { useState } from 'react';
import { View, Text, StyleSheet, TextInput, Button, Alert } from 'react-native';

const ProfileScreen = () => {
  // Sample user data
  const [username, setUsername] = useState('JohnDoe');
  const [email, setEmail] = useState('johndoe@example.com');

  const handleEditProfile = () => {
    // Handle the profile editing logic here
    Alert.alert('Profile Edited', 'Your profile has been updated successfully!');
  };

  const handleLogout = () => {
    // Handle logout logic here
    Alert.alert('Logged Out', 'You have been logged out successfully!');
  };

  return (
    <View style={styles.screenContainer}>
      <Text style={styles.title}>Profile Screen</Text>
      
      <View style={styles.inputContainer}>
        <Text style={styles.label}>Username:</Text>
        <TextInput
          style={styles.input}
          value={username}
          onChangeText={setUsername}
        />
      </View>

      <View style={styles.inputContainer}>
        <Text style={styles.label}>Email:</Text>
        <TextInput
          style={styles.input}
          value={email}
          onChangeText={setEmail}
          keyboardType="email-address"
        />
      </View>

      <Button title="Edit Profile" onPress={handleEditProfile} color="rgb(174,12,46)" />
      <Button title="Logout" onPress={handleLogout} color="#FF5722" style={styles.logoutButton} />
    </View>
  );
};
